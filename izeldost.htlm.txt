        function celebrate() {
            document.getElementById('celebration').style.display = 'flex';
            
            // Create massive confetti
            for(let i = 0; i < 100; i++) {
                setTimeout(() => createConfetti(), i * 30);
            }

            // Fireworks effect
            setTimeout(() => {
                for(let i = 0; i < 5; i++) {
                    setTimeout(() => createFirework(), i * 500);
                }
            }, 500);
        }

        function createConfetti() {
            const confetti = document.createElement('div');
            const emojis = ['‚ù§Ô∏è', 'üíï', 'üíñ', 'üíó', 'üíù', '‚ú®', 'üåü', '‚≠ê', 'üí´', 'üéâ', 'üéä', 'üåπ'];
            confetti.innerHTML = emojis[Math.floor(Math.random() * emojis.length)];
            confetti.style.position = 'fixed';
            confetti.style.left = Math.random() * 100 + '%';
            confetti.style.top = '-50px';
            confetti.style.fontSize = (Math.random() * 20 + 20) + 'px';
            confetti.style.zIndex = '9999';
            confetti.style.pointerEvents = 'none';
            document.body.appendChild(confetti);

            let fall = -50;
            let swing = Math.random() * 200 - 100;
            const fallSpeed = Math.random() * 3 + 3;
            const swingSpeed = (Math.random() - 0.5) * 2;

            const fallInterval = setInterval(() => {
                fall += fallSpeed;
                swing += swingSpeed;
                confetti.style.top = fall + 'px';
                confetti.style.left = `calc(${confetti.style.left} + ${swing}px)`;
                confetti.style.transform = `rotate(${fall * 2}deg)`;
                
                if(fall > window.innerHeight + 50) {
                    clearInterval(fallInterval);
                    confetti.remove();
                }
            }, 20);
        }

        function createFirework() {
            const centerX = Math.random() * window.innerWidth;
            const centerY = Math.random() * window.innerHeight * 0.7;

            for(let i = 0; i < 20; i++) {
                const particle = document.createElement('div');
                particle.innerHTML = '‚ú®';
                particle.style.position = 'fixed';
                particle.style.left = centerX + 'px';
                particle.style.top = centerY + 'px';
                particle.style.fontSize = '25px';
                particle.style.zIndex = '9999';
                particle.style.pointerEvents = 'none';
                document.body.appendChild(particle);

                const angle = (Math.PI * 2 * i) / 20;
                const velocity = Math.random() * 5 + 5;
                let x = centerX;
                let y = centerY;
                let vx = Math.cos(angle) * velocity;
                let vy = Math.sin(angle) * velocity;

                const moveInterval = setInterval(() => {
                    x += vx;
                    y += vy;
                    vy += 0.2; // gravity
                    particle.style.left = x + 'px';
                    particle.style.top = y + 'px';
                    particle.style.opacity = parseFloat(particle.style.opacity || 1) - 0.02;

                    if(parseFloat(particle.style.opacity) <= 0 || y > window.innerHeight) {
                        clearInterval(moveInterval);
                        particle.remove();
                    }
                }, 20);
            }
        }
